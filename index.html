function selectLanguage(lang) { currentLang = lang; const htmlElement = document.getElementById('htmlElement'); document.body.classList.add('language-selected'); document.getElementById('languageOverlay').remove(); htmlElement.dir = lang === 'ar' ? 'rtl' : 'ltr'; htmlElement.lang = lang; document.body.classList.toggle('english-font', lang === 'en'); document.querySelectorAll('[data-translate-key]').forEach(element => { const key = element.dataset.translateKey; element.textContent = translations[key][lang]; }); document.getElementById('langSwitcher').textContent = lang === 'ar' ? 'English' : 'العربية'; showEmployeeGrid(); } function toggleLanguage() { currentLang = currentLang === 'ar' ? 'en' : 'ar'; const htmlElement = document.getElementById('htmlElement'); htmlElement.dir = currentLang === 'ar' ? 'rtl' : 'ltr'; htmlElement.lang = currentLang; document.getElementById('langSwitcher').textContent = currentLang === 'ar' ? 'English' : 'العربية'; document.body.classList.toggle('english-font', currentLang === 'en'); document.querySelectorAll('[data-translate-key]').forEach(element => { const key = element.dataset.translateKey; element.textContent = translations[key][currentLang]; }); if (document.querySelector('.employee-grid')) { showEmployeeGrid(); } else { renderCards(); } }
